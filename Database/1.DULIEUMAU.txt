-- Ví dụ bảng NHANVIEN với mã nhân viên tự động
CREATE TABLE NHANVIEN(
	   IDNV UNIQUEIDENTIFIER PRIMARY KEY DEFAULT NEWID(),
	   MANV AS ('NV'+ RIGHT('000' + CAST(IDNV AS VARCHAR(3)), 3)) PERSISTED UNIQUE, -- Tự sinh mã nhân viên
	   TENNV NVARCHAR(50),
	   GIOITINH BIT, -- 1.NAM || 0.NU
	   NGAYSINH DATE,
	   CCCD VARCHAR(13) UNIQUE,
	   DIACHI NVARCHAR(MAX),
	   SDT VARCHAR(13) UNIQUE,
	   TRANGTHAINV BIT, -- 1.LAM || 0.NGHI
	   CHUCVU NVARCHAR(20),
	   MATKHAU VARCHAR(20)
);
